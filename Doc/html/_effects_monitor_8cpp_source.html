68 void EffectsMonitor::loadFiles() {
69  this->effects.clear();
70  this->files.clear();
71  std::map<Effect::effectID, std::string> effects = Effect::getEffects();
72  QStringList files = QDir(QString::fromStdString(folder)).entryList(
73  QStringList("*.xml"));
74  for (int index = 0; index < files.size(); index++) {
75  QFile *file = new QFile(QString::fromStdString(folder) + files[index]);
76  if (file->open(QFile::ReadOnly | QFile::Text)) {
77  QTextStream *in = new QTextStream(file);
78  std::string xml = in->readAll().toStdString();
79  file->close();
80  if (xml.find("<!DOCTYPE fxtemplate SYSTEM \"") < xml.size()) {
81  std::string descriptor = "<!DOCTYPE fxtemplate SYSTEM \"";
82  std::string reference = xml.substr(
83  xml.find(descriptor) + descriptor.size());
84  reference = reference.substr(0, reference.find("\""));
85  while (reference.size() > 0) {
86  std::string effectname = reference.substr(0,
87  reference.find(","));
88  if (reference.find(",") < reference.size()) {
89  reference = reference.substr(reference.find(",") + 1);
90  } else {
91  reference.clear();
92  }
93  while (reference[0] == ' ') {
94  reference = reference.substr(1);
95  }
96  Effect::effectID effect = Effect::getEffect(effectname);
97  if (this->effects.find(effect) == this->effects.end()) {
98  this->effects.insert(
99  std::pair<Effect::effectID, std::string>(effect,
100  effects[effect]));
101  this->files.insert(
102  std::pair<Effect::effectID, std::string>(effect,
103  file->fileName().toStdString()));
104  consolelog("EffectsMonitor", LogType::info,
105  "available effect \"" + effectname
106  + "\" was found in \""
107  + file->fileName().toStdString()
108  + "\"");
109  } else {
110  consolelog("EffectsMonitor", LogType::warning,
111  "available effect \"" + effectname
112  + "\" was found again in \""
113  + file->fileName().toStdString()
114  + "\" so this last reference will be ignored");
115  }
116  }
117  } else {
118  consolelog("EffectsMonitor", LogType::warning,
119  "\"" + file->fileName().toStdString()
120  + "\" does not have a valid template descriptor so it will be ignored");
121  }
122  } else {

132 void EffectsMonitor::loadTemplate() {
133  this->clear();
134  QFile *file = new QFile(
135  QString::fromStdString(this->files[this->effect->effect.first]));
136  QTextStream *in = new QTextStream(file);
137  if (file->open(QFile::ReadOnly | QFile::Text)) {
138  std::string xml = in->readAll().toStdString();
139  file->close();
140  xml = xml.substr(xml.find(">") + 1); // deleting XML descriptor
141  while (xml.find("<") < xml.size()) {
142  // Tag
143  std::string element = xml.substr(xml.find("<"));
144  std::string tag = element.substr(1, element.find(" ") - 1);
145  element = element.substr(0,
146  element.find("</" + tag + ">") + tag.size() + 3);
147  this->loadField(element);
148  if (xml.find("</" + tag + ">") + tag.size() + 3 < xml.size()) {
149  xml = xml.substr(xml.find("</" + tag + ">") + tag.size() + 3);
150  } else {
151  xml.clear();
152  }
153  }
154  consolelog("EffectsMonitor", LogType::progress,
155  "template file has been loaded");
156  } else {
157  consolelog("EffectsMonitor", LogType::error,
158  "\"" + file->fileName().toStdString()
159  + "\" could not be opened");

167 void EffectsMonitor::loadField(std::string element) {
168  const int width = 100;
169  std::string tag = element.substr(1, element.find(" ") - 1);
170  // Attributes
171  std::string id = this->getAttribute(element, "id");
172  // Deleting banned characters for id
173  std::string newid = "";
174  for (int character = 0; character < (int) id.size(); character++) {
175  bool valid = false;
176  if (character > 0) {
177  valid += (0x30 <= id[character] && id[character] <= 0x39); // numbers
178  }
179  valid += (0x41 <= id[character] && id[character] <= 0x5A); // uppercase letters
180  valid += (0x61 <= id[character] && id[character] <= 0x7A); // lowercase letters
181  if (valid) {
182  newid += id[character];
183  } else {
184  consolelog("EffectsMonitor", LogType::warning,
185  "character \"" + id.substr(character, 1)
186  + "\" is not allowed as parameter name in \""
187  + id + "\" and it will be removed");
188  }
189  }
190  id = newid;
191  std::string text = this->getAttribute(element, "text");
192  std::string value = this->getAttribute(element, "value");
193  std::string suffix = this->getAttribute(element, "suffix");
194  bool readonly = Effect::getBool(this->getAttribute(element, "readonly"));
195  double min = atof(this->getAttribute(element, "min").c_str());
196  double max = std::atof(this->getAttribute(element, "max").c_str());
197  double step = std::atof(this->getAttribute(element, "step").c_str());
198  if (step <= 0) {
199  step = 1;
200  }
201  int length = std::atoi(this->getAttribute(element, "length").c_str());
202  if (length <= 0 || length > 100) {
203  length = 100;
204  }
205  // Contain
206  std::string contain = element.substr(element.find(">") + 1);
207  if (contain.rfind("<")==0) {
208  contain = "";
209  } else {
210  contain = contain.substr(0,contain.rfind("<"));
211  }
212  // User interface
213  if (tag == "int") {
214  QSpinBox *field = new QSpinBox(this->framework);
215  field->setMinimumWidth(width);
216  field->setObjectName(QString::fromStdString(prefix + id));
217  this->parameters.insert(std::pair<std::string, std::string>(id, value));
218  field->setReadOnly(readonly);
219  field->setSuffix(QString::fromStdString(suffix));
220  field->setMinimum(min);
221  field->setMaximum(max);
222  field->setValue(std::atoi(value.c_str()));
223  field->setSingleStep(step);
224  this->layout->addRow(QString::fromStdString(text), field);
225  QObject::connect(field,SIGNAL(valueChanged(int)),this,SLOT(updateParameter(int)));
226  } else if (tag == "double") {
227  QDoubleSpinBox *field = new QDoubleSpinBox(this->framework);
228  field->setMinimumWidth(width);
229  field->setObjectName(QString::fromStdString(prefix + id));
230  this->parameters.insert(std::pair<std::string, std::string>(id, value));
231  field->setReadOnly(readonly);
232  field->setSuffix(QString::fromStdString(suffix));
233  field->setMinimum(min);
234  field->setMaximum(max);
235  field->setValue(std::atof(value.c_str()));
236  field->setSingleStep(step);
237  this->layout->addRow(QString::fromStdString(text), field);
238  QObject::connect(field,SIGNAL(valueChanged(double)),this,SLOT(updateParameter(double)));
239  } else if (tag == "string") {
240  QLineEdit *field = new QLineEdit(this->framework);
241  field->setMinimumWidth(width);
242  field->setObjectName(QString::fromStdString(prefix + id));
243  this->parameters.insert(std::pair<std::string, std::string>(id, value));
244  field->setText(QString::fromStdString(value));
245  field->setReadOnly(readonly);
246  field->setMaxLength(length);
247  this->layout->addRow(QString::fromStdString(text), field);
248  QObject::connect(field, SIGNAL(textChanged(QString)), this,
249  SLOT(updateParameter(QString)));
250  } else if (tag == "bool") {
251  QCheckBox *field = new QCheckBox(this->framework);
252  field->setMinimumWidth(width);
253  field->setObjectName(QString::fromStdString(prefix + id));
254  this->parameters.insert(std::pair<std::string, std::string>(id, value));
255  field->setChecked(
256  value == "true" || value == "True" || value == "TRUE" || value == "1");
257  field->setEnabled(!readonly);

259  QObject::connect(field,SIGNAL(toggled(bool)),this,SLOT(updateParameter(bool)));
260  } else if (tag == "enum") {

301  field->setObjectName(QString::fromStdString(prefix + id));
302  field->setText(QString::fromStdString(text));
303  this->layout->addRow(field);

306  field->setMinimumSize(300, 225);
307  field->setObjectName(QString::fromStdString(prefix + id));
308  double range[2][2] = {{Effect::getDouble(this->getAttribute(element, "xmin")),
309  Effect::getDouble(this->getAttribute(element, "xmax"))},{
310  Effect::getDouble(this->getAttribute(element, "ymin")),
311  Effect::getDouble(this->getAttribute(element, "ymax"))}};
312  std::string xlabel = this->getAttribute(element, "xlabel");
313  std::string ylabel = this->getAttribute(element, "ylabel");
314  int options = 0;
315  if (xlabel != "") {
316  options |= Chart2D::ChartOptions::labelX;
317  }
318  if (ylabel != "") {
319  options |= Chart2D::ChartOptions::labelY;
320  }
321  if (this->getAttribute(element, "xscale") == "log") {
322  options |= Chart2D::ChartOptions::logX;
323  }
324  if (this->getAttribute(element, "ylabel") == "log") {
325  options |= Chart2D::ChartOptions::logY;
326  }
327  Chart2D *chart = new Chart2D(field, range, text, xlabel, ylabel, options);
328  charts.insert(std::pair<std::string, Chart2D *>(id, chart));
329  this->layout->addRow(field);
330  } else {

332  "tag \"" + tag + "\" is not recognised as a valid tag");
333  }
334  consolelog("EffectsMonitor", LogType::progress, "element \"" + id + "\" of type \"" + tag + "\" was created");
335 }
