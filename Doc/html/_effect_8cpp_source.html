40 void Effect::setParams(std::map<std::string, std::string> params) {
41  this->params.clear();
42  // Deleting banned characters for param
43  for (std::map<std::string, std::string>::iterator iterator = params.begin(); iterator != params.end(); iterator++) {
44  std::string param = "";
45  for (int character = 0; character < (int)iterator->first.size(); character++) {
46  bool valid = false;
47  if (character > 0) {
48  valid += (0x30 <= iterator->first[character] && iterator->first[character] <= 0x39); // numbers
49  }
50  valid += (0x41 <= iterator->first[character] && iterator->first[character] <= 0x5A); // uppercase letters
51  valid += (0x61 <= iterator->first[character] && iterator->first[character] <= 0x7A);// lowercase letters
52  if (valid) {
53  param += iterator->first[character];
54  } else {
55  consolelog("Effect", LogType::warning, "character \"" + iterator->first.substr(character,1) + "\" is not allowed as parameter name in \"" + iterator->first + "\" and it will be removed");
56  }
57  }
58  this->params.insert(std::pair<std::string, std::string>(param, iterator->second));
59  }

224 std::map<std::string, std::string> Effect::getTagMap(std::string configuration, std::string tag) {
225  configuration = Effect::getTag(configuration, tag);
226  std::map<std::string, std::string> map;
227  while ((int) configuration.size() > 0) {
228  std::string line = configuration.substr(0, configuration.find(";"));
229  if (configuration.find(";") < configuration.size()) {
230  configuration = configuration.substr(configuration.find(";") + 1);
231  } else {
232  configuration = "";
233  }
234  // Control characters deletion
235  std::string newline = "";
236  for (int character = 0; character < (int) line.size(); character++) {
237  if (line[character] >= 0x20) {
238  newline += line[character];
239  }
240  }
241  line = newline;
242  if (line.find("=") < line.size()) {
243  // Key
244  std::string key = line.substr(0, line.find("="));
245  while (key[0] == ' ') {
246  key = key.substr(1);
247  }
248  while (key[key.size() - 1] == ' ') {
249  key = key.substr(0, key.size() - 1);
250  }
251  // Value
252  std::string value = line.substr(line.find("=") + 1, line.find(";") - line.find("="));
253  // Spaces deletion at the beggining
254  while (value[0] == ' ') {
255  value = value.substr(1);
256  }
257  // Spaces deletion at the end
258  while (value[value.size() - 1] == ' ') {
259  value = value.substr(0, value.size() - 1);
260  }
261  map.insert(std::pair<std::string, std::string>(key, value));
262  } else if (line != "") {
263  consolelog("Effect", LogType::error, "syntax error in \"" + line + "\" (\"=\" is missed)");
264  }
265  }
266  return map;
267 }
