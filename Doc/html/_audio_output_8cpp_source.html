195 void OutputDevice::send(float *signal, int samples) {
196  int sample_bytes = this->format.sampleSize() / 8; // sample size [Bytes]
197  int bytes = samples * sample_bytes; // number of bytes to be sent [Bytes]
198  double value = 0;
199  char data;
200  for (int sample = 0; sample < samples; sample++) {
201  value = (offset + signal[sample]) * amplitude;
202  if (this->format.sampleType() == QAudioFormat::UnSignedInt) {
203  value /= 2;
204  }
205  for (int byte = 0; byte < sample_bytes; byte++) {
206  switch (this->format.byteOrder()) {
207  case QAudioFormat::LittleEndian:
208  data = (int) value >> byte * 8;
209  break;
210  case QAudioFormat::BigEndian:
211  default:
212  data = (int) value >> (sample_bytes - byte - 1) * 8;
213  break;
214  }
215  this->buffer[(this->cursor_write + sample * sample_bytes + byte)
216  % this->buffersize] = data;
217  }
218  }
219  this->cursor_write = (this->cursor_write + bytes) % this->buffersize;
220 }

228 qint64 OutputDevice::readData(char *data, qint64 length) {
229  int chunk = 0;
230  if (this->cursor_read < this->cursor_write) {
231  chunk = std::min((int) length, this->cursor_write - this->cursor_read);
232  std::memcpy(data, this->buffer + this->cursor_read, chunk);
233  } else if (this->cursor_read > this->cursor_write) {
234  chunk = std::min((int) length,
235  this->cursor_write - this->cursor_read + this->buffersize);
236  if (this->cursor_read + chunk <= this->buffersize) {
237  std::memcpy(data, this->buffer + this->cursor_read, chunk);
238  } else {
239  std::memcpy(data, this->buffer + this->cursor_read,
240  this->buffersize - this->cursor_read); // until the buffer end
241  std::memcpy(data, this->buffer,
242  chunk - (this->buffersize - this->cursor_read)); // until the buffer cursor
243  }
244  }
245  this->cursor_read = (this->cursor_read + chunk) % this->buffersize;
246  return chunk;
247 }
