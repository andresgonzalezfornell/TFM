127 bool ProcessManager::applyEffect(Effect *effect, std::vector<bool> channels,
128  std::vector<double> levels) {
129  if (this->cursor < this->samples) {
130  if ((int) channels.size() == this->channels) {
131  int n, N;
132  if (this->chunksize == 0) {
133  n = 0;
134  N = this->samples;
135  } else {
136  n = this->cursor;
137  N = this->chunksize;
138  if (n + N >= this->samples) {
139  N = this->samples - n;
140  }
141  }
142  float **input_effect = (float **) std::malloc(this->channels * sizeof(float *));
143  float **output_effect = (float **) std::malloc(this->channels * sizeof(float *));
144  for (int channel = 0; channel < this->channels; channel++) {
145  input_effect[channel] = (float *) std::malloc(N * sizeof(float));
146  output_effect[channel] = (float *) std::malloc(N * sizeof(float));
147  // Pre-amplifier
148  switch (this->bitstream->channels[channel]) {
149  case SACBitstream::ChannelType::Ls:
150  case SACBitstream::ChannelType::Lsr:
151  case SACBitstream::ChannelType::Rs:
152  case SACBitstream::ChannelType::Rsr:
153  levels[channel] *= this->bitstream->gain_surround;
154  break;
155  case SACBitstream::ChannelType::LFE:
156  levels[channel] *= this->bitstream->gain_LFE;
157  break;
158  default:
159  break;
160  }
161  // Levels
162  for (int sample = n; sample < (n + N); sample++) {
163  input_effect[channel][sample - n] = levels[channel] * this->input[channel][sample];
164  }
165  }
166  // Effect
167  effect->apply(input_effect, output_effect, N, this->bitstream->channels);
168  for (int channel = 0; channel < this->channels; channel++) {
169  // Channels selection
170  if (channels[channel]) {
171  std::memcpy(this->output[channel] + n, output_effect[channel], N*sizeof(float));
172  } else {
173  std::memcpy(this->output[channel] + n, input_effect[channel], N*sizeof(float));
174  }
175  // Post-amplifier
176  switch (this->bitstream->channels[channel]) {
177  case SACBitstream::ChannelType::Ls:
178  case SACBitstream::ChannelType::Lsr:
179  case SACBitstream::ChannelType::Rs:
180  case SACBitstream::ChannelType::Rsr:
181  levels[channel] /= this->bitstream->gain_surround;
182  for (int sample = n; sample < (n + N); sample++) {
183  this->output[channel][sample] /= this->bitstream->gain_surround;
184  }
185  break;
186  case SACBitstream::ChannelType::LFE:
187  levels[channel] /= this->bitstream->gain_LFE;
188  for (int sample = n; sample < (n + N); sample++) {
189  this->output[channel][sample] /= this->bitstream->gain_LFE;
190  }
191  break;
192  default:
193  break;
194  }
195  }
196  std::free(input_effect);
197  std::free(output_effect);
198  this->cursor += N;
199  if (this->cursor > this->total) {
200  this->total = this->cursor;
201  }
202  return true;
203  } else {
204  consolelog("ProcessManager", LogType::error,
205  "channels boolean vector (from effect argument) = "
206  + std::to_string(channels.size())
207  + " does not correspond to number of channels = "
208  + std::to_string(this->channels));
209  return false;
210  }
211  } else {
212  consolelog("ProcessManager", LogType::error, "there is no more available samples (samples = " + std::to_string(this->samples) + ")");
213  return false;
214  }
215 }
