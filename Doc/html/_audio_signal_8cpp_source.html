197  int F = spectrum.size();
198  if (F <= bands) {
199  return spectrum;
200  } else {
201  std::vector<float> spectrum_shrunk = std::vector<float>(bands);
202  int step = ceil(F / bands);
203  int band = 0;
204  for (int f = 0; f < F; f += step) {
205  float band_level = 0;
206  int substeps;
207  if (f + step <= F) {
208  substeps = f + step;

210  substeps = F;
211  }
212  // Band level arithmetic mean calculation
213  for (int substep = f; substep < substeps; substep++) {
214  band_level += spectrum[substep] / step;
