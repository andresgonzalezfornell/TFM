442 void SACEffects::sendOutput() {
443  int N; // number of samples to be plotted
444  int n; // first sample to be plotted
445  if (this->chunksize == 0) {
446  N = this->process->samples;
447  n = 0;
448  } else {
449  N = this->chunksize;
450  n = this->process->cursor - N;
451  if (n + N >= this->process->samples) {
452  N = this->process->samples - n - 1;
453  }
454  }
455  for (int channel = 0; channel < this->channels_output->getSize(); channel++) {
456  if (!muted) {
457  this->channels_output->getChannel(channel)->audiooutput->outputdevice->send(this->process->output[channel] + n, N);
458  }
459  }
460  this->setTimer();
461  this->applyEffect();
462 }

514 void SACEffects::setTimer() {

516  double time = (double) 1000 * this->process->cursor / this->fs; // current playback time [ms]
517  int h = (int) std::floor(time / 1000 / 60 / 60);
518  int m = (int) std::floor(time / 1000 / 60) % 60;
519  int s = (int) std::floor(time / 1000) % 60;
520  int ms = (int) std::round(time) % 1000;
521  ui->input_timer->blockSignals(true);
522  ui->output_timer->blockSignals(true);
523  ui->input_timer->setTime(QTime(h, m, s, ms));
524  ui->output_timer->setTime(QTime(h, m, s, ms));
525  ui->input_timer->blockSignals(false);
526  ui->output_timer->blockSignals(false);
527  if (this->process->cursor == this->process->total) {
528  this->setDuration(ui->output_duration, (double) time / 1000);
529  }
530  } else {
531  this->stop();
532  }
