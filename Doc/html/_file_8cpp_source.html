424 float **WAVFile::readSamples(int samples) {
425  if (samples > 0) {
426  int bytes = this->header.bitspersample / 8;
427  int channels = this->header.numchannels;
428  float **array = (float **)std::malloc(channels * sizeof(float *));
429  for(int channel = 0; channel < channels; channel++) {
430  array[channel] = (float *)std::malloc(samples * sizeof(float));
431  }
432  unsigned int amplitude = 0x1 << (this->header.bitspersample - 1);
433  unsigned int signmask = amplitude;
434  unsigned int valuemask = signmask - 1;
435  unsigned int number = 0;
436  int value = 0;
437  char *data = this->read(bytes * channels * samples);
438  for(int sample = 0; sample < samples; sample++) {
439  for (int channel = 0; channel < channels; channel++) {
440  number = 0;
441  for (int byte = 0; byte < bytes; byte++) {
442  number |= (unsigned char) data[byte + bytes*(channel + channels*sample)] << (8 * byte);
443  }
444  bool negative = (number & signmask) != 0;
445  if (negative) {
446  value = (~number) & valuemask; // inverting bits
447  value += 1; // 2's complement
448  value *= -1; // sign
449  } else {
450  value = number & valuemask;
451  }
452  array[channel][sample] = (float) value / amplitude;
453  }
454  }
455  return array;
456  } else {
457  consolelog("WAVFile", LogType::error, "number of samples to read (" + std::to_string(samples) + ")should be greater than 0");
458  return NULL;
459  }

468  if (samples > 0) {
469  int bytes = this->header.bitspersample / 8;
470  int channels = this->header.numchannels;
471  char *data = (char *)std::malloc(channels * samples * bytes);
472  unsigned int amplitude = 0x1 << (this->header.bitspersample - 1);
473  int value = 0;
474  for(int sample = 0; sample < samples; sample++) {
475  for (int channel = 0; channel < channels; channel++) {
476  value = array[channel][sample] * amplitude;
477  // Solving value according to preccision
478  if ((unsigned int)value >= amplitude && array[channel][sample] > 0) {
479  value = amplitude - 1;

481  value = - amplitude;
482  }
483  for (int byte = 0; byte < bytes; byte++) {
484  data[byte + bytes*(channel + channels*sample)] = value >> (8 * byte);
485  }
486  }
487  }
