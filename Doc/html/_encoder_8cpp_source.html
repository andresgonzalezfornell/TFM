134 void Encoder::setBitstream() {
135  QObject::sender()->blockSignals(true);
136  std::string filename = ui->bitstream_filename->text().toStdString();
137  // Deleting reserved characters
138  char reservedchar[6] = { '/', '>', '<', '|', ':', '&' };
139  for (int index = 0; index < (int) sizeof(reservedchar); index++) {
140  while (filename.find(reservedchar[index]) < filename.length()) {
141  filename = filename.substr(0, filename.find(reservedchar[index]))
142  + filename.substr(filename.find(reservedchar[index]) + 1);
143  }
144  }
145  ui->bitstream_filename->setText(QString::fromStdString(filename));
146  consolelog("Encoder", LogType::interaction,
147  "bitstream file name changed to " + filename);
148  QObject::sender()->blockSignals(false);
149 }

216 void Encoder::apply() {
217  QObject::sender()->blockSignals(true);
218  consolelog("Encoder", LogType::interaction, "encoding");
219  std::string input_str = this->input->getFilename();
220  std::string output_str = QFileDialog::getSaveFileName(NULL,
221  "Save output file for encoder", "", "*.wav").toStdString();
222  std::string bitstream_str = output_str.substr(0,
223  output_str.rfind("/") + 1);
224  if (output_str != "") {
225  // SAC encoding
226  consolelog("Encoder", LogType::progress, "proceeding to SAC encoder");
227  if (ui->bitstream_buried->isChecked()) {
228  bitstream_str = "buried";
229  } else {
230  bitstream_str += ui->bitstream_filename->text().toStdString();
231  }
232  const char *input_char = input_str.c_str();
233  const char *bitstream_char = bitstream_str.c_str();
234  const char *output_char = output_str.c_str();
235  int timeslots = this->input->header.bitspersample;
236  consolelog("Encoder", LogType::info, "input file:\t " + input_str);
237  consolelog("Encoder", LogType::info,
238  "bitstream file:\t " + bitstream_str);
239  consolelog("Encoder", LogType::info, "output file:\t " + output_str);
240  consolelog("Encoder", LogType::info,
241  "sampling frequency:\t " + std::to_string(this->fs) + "Hz");
242  consolelog("Encoder", LogType::info,
243  "tree configuration:\t " + std::to_string(this->tree));
244  consolelog("Encoder", LogType::info,
245  "time slots:\t " + std::to_string(timeslots) + "bits");
246  char *error = sac_encode(input_char, output_char, bitstream_char,
247  (double) this->fs, this->tree, timeslots); // SAC function
248  if (error == NULL) {
249  this->setOutput(output_str);
250  if (bitstream_str == "buried") {
251  this->bitstream = NULL;
252  } else {
253  this->bitstream = new File(bitstream_str, false);
254  }
255  consolelog("sac_encoder", LogType::progress,
256  "encoding was completed successfully");
257  } else {
258  consolelog("sac_encoder", LogType::error, std::string(error));
259  }
260  }
261  QObject::sender()->blockSignals(false);
262 }
