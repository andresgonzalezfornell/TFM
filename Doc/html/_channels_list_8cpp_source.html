474 void ChannelsCharts::plot() {
475  QVector < QPointF > points;
476  float *samples;
477  Chart2D *chart;
478  int remainder = 0;
479  double range[2][2];
480  for (int plot = 0; plot < 2; plot++) {
481  switch (plot) {
482  case 0:
483  // Input
484  samples = this->input[(int)std::floor(ui->input_selector->currentIndex() / 2)];
485  remainder = (int)ui->input_selector->currentIndex() % 2;
486  chart = this->input_chart;
487  break;
488  case 1:
489  // Output
490  samples = this->output[(int)std::floor(ui->output_selector->currentIndex() / 2)];
491  remainder = (int)ui->output_selector->currentIndex() % 2;
492  chart = this->output_chart;
493  break;
494  }
495  switch (remainder) {
496  case 0:
497  // Signal plot
498  range[0][0] = (double) this->getTimeCursor() / ChannelsList::fs;
499  range[0][1] = (double) (this->getTimeCursor() + this->getScope() - 1) / ChannelsList::fs ;
500  range[1][0] = -1;
501  range[1][1] = 1;
502  chart->setOptions(Chart2D::ChartOptions::labelX);
503  for (int sample = this->getTimeCursor(); sample < this->getTimeCursor() + this->getScope(); sample++) {
504  points.push_back(QPointF((float) sample / ChannelsList::fs, samples[sample]));
505  }
506  break;
507  case 1:
508  // Spectrum plot
509  std::vector<float> vector = std::vector<float>(samples + this->getTimeCursor(), samples + this->getTimeCursor() + this->getScope());
510  AudioSignal signal = AudioSignal(vector, ChannelsList::fs);
511  range[0][0] = 0;
512  range[0][1] = (double) ChannelsList::fs / 2;
513  range[1][0] = 0;
514  range[1][1] = 1;
515  chart->setOptions(Chart2D::ChartOptions::labelX);
516  std::vector<float> spectrum = signal.getSpectrum();
517  for (int frequency = 0; frequency < std::ceil((double) this->getScope() / 2); frequency++) {
518  points.push_back(QPointF((float) frequency * ChannelsList::fs / this->getScope(), spectrum[frequency]));
519  }
520  break;
521  }
522  chart->clear();
523  chart->setRange(range);
524  chart->setPoints(points);
525  points.clear();
526  }
