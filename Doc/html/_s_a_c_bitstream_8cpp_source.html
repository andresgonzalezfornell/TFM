43 long int SACBitstream::getVariable(int position, int length) {
44  long int value = 0;
45  int byte = position / 8;
46  int bit = position % 8;
47  int bytes = std::ceil((double)(bit + length) / 8);
48  int bits = (bit + length) % 8;
49  if (bits == 0) {
50  bits = 8;
51  }
52  this->bitstream->setCursor(byte);
53  std::string data = bitstream->readText(bytes);
54  for (int index = 0; index < bytes; index++) {
55  char mask = 0xFF;
56  if (index == 0) {
57  mask = pow(2, 8 - bit) - 1;
58  }
59  char databyte;
60  if(index == bytes - 1) {
61  databyte = (data[index] & mask) >> (8 - bits);
62  value = value << bits;
63  } else {
64  databyte = data[index] & mask;
65  value = value << 8;
66  }
67  value |= databyte;
68  }
69  return value;
70 }
